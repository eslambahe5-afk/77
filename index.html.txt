    <script>
        // تحميل بيانات المستخدم من الذاكرة
        window.onload = function() {
            checkLogin();
        };

        // فحص حالة تسجيل الدخول
        function checkLogin() {
            let userData = localStorage.getItem('loggedUser');
            if (userData) {
                userData = JSON.parse(userData);
                showUserPanel(userData);
            }
        }

        // تسجيل الدخول
        document.querySelector('form').onsubmit = function(e) {
            e.preventDefault();
            
            let email = document.getElementById('userEmail').value;
            let password = document.getElementById('userPassword').value;
            let remember = document.getElementById('remember').checked;

            // التحقق من البيانات
            if (!email || !password) {
                showAlert('الرجاء إدخال جميع البيانات', 'error');
                return;
            }

            if (!email.includes('@')) {
                showAlert('البريد الإلكتروني غير صحيح', 'error');
                return;
            }

            if (password.length < 4) {
                showAlert('كلمة المرور قصيرة جداً', 'error');
                return;
            }

            // حفظ البيانات
            let userData = {
                email: email,
                time: new Date().toLocaleString('ar-EG'),
                remember: remember
            };

            localStorage.setItem('loggedUser', JSON.stringify(userData));
            showUserPanel(userData);
            showAlert('تم تسجيل الدخول بنجاح', 'success');
        };

        // إظهار لوحة المستخدم
        function showUserPanel(data) {
            document.getElementById('loginForm').style.display = 'none';
            document.getElementById('userPanel').classList.add('show');
            
            document.getElementById('displayName').textContent = data.email.split('@')[0];
            document.getElementById('displayEmail').textContent = data.email;
            document.getElementById('loginTime').textContent = data.time;
        }

        // تسجيل الخروج
        function logout() {
            localStorage.removeItem('loggedUser');
            document.getElementById('loginForm').style.display = 'block';
            document.getElementById('userPanel').classList.remove('show');
            document.querySelector('form').reset();
            showAlert('تم تسجيل الخروج', 'success');
        }

        // عرض رسائل التنبيه
        function showAlert(msg, type) {
            let alertEl = document.getElementById('alertMsg');
            alertEl.textContent = msg;
            alertEl.className = 'message ' + type;
            alertEl.style.display = 'block';
            
            setTimeout(function() {
                alertEl.style.display = 'none';
            }, 3000);
        }

        // نسيت كلمة المرور
        function forgotPass() {
            let email = document.getElementById('userEmail').value;
            if (email) {
                showAlert('تم إرسال رابط إعادة التعيين لبريدك', 'success');
            } else {
                showAlert('أدخل بريدك الإلكتروني أولاً', 'error');
            }
        }
